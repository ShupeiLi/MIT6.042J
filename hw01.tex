\section*{Problem 1}
\subsection*{(a)}
\begin{align*}
    \exists x, S(x)\land A(x)
\end{align*}

\subsection*{(b)}
\begin{align*}
    \forall x, T(x)\land A(x)
\end{align*}
{\color{red}\textbf{Solution: }
\begin{align*}
    \forall x\in X: T(x) \land S(x) \Rightarrow A(x)
\end{align*}
}

\subsection*{(c)}
\begin{align*}
    \neg \left( \exists x, T(x)\land \left( \neg A(x) \right)  \right) 
\end{align*}
{\color{blue}\textbf{Solution: }
\begin{align*}
    \neg\exists x\in X: T(x)\land \left( \neg A(x) \right) 
\end{align*}
}

\subsection*{(d)}
\begin{align*}
    \exists x, y, z, x \neq y, y \neq z, x \neq z, T(u)\land \left( \neg S(u) \right), u\in \left\{x, y, z \right\} 
\end{align*}
{\color{blue}\textbf{Solution: }
\begin{align*}
    \exists x, y, z\in X: \left(\neg E(x, y)\land \neg E(y, z) \land\neg E(x, z) \right) \land T(x) \land \neg S(x) \land T(y) \land \neg S(y) \land T(z) \land \neg S(z)
\end{align*}
}

\section*{Problem 3}
\subsection*{(a)}
\subsubsection*{(1)}
\begin{align*}
    A \land B = \neg \left( A \barwedge B \right) 
\end{align*}

\subsubsection*{(2)}
\begin{align*}
   A \lor B = \left( \neg A \right)  \barwedge \left( \neg B \right) 
\end{align*}

\subsubsection*{(3)}
\begin{align*}
    A \Rightarrow B = A \barwedge \left( A \barwedge B \right) 
\end{align*}
{\color{blue}\textbf{Solution: }
\begin{align*}
    A\Rightarrow B &= \left( \left( \neg A \right)\lor B \right)\\
                   &= \neg \left( A\land \left( \neg B \right)  \right) \\
                   &= A \barwedge \left( \neg B \right) 
\end{align*}
}

\subsection*{(b)}
\begin{align*}
    \neg A = A \barwedge A
\end{align*}

\subsection*{(c)}
\begin{align*}
    \text{true} &= A \barwedge \left( A \barwedge A \right)\\ 
    \text{false} &= \left( A \barwedge \left( A \barwedge A \right)  \right) \barwedge \left( A \barwedge \left( A \barwedge A \right)  \right)
\end{align*}
